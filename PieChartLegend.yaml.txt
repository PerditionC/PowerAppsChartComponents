    Properties:
      Height: =400
      OnReset: |-
        =// --- DEV: make the debug Reset button visible so it can be used during authoring ---
        Set(debugResetVisible, true);

        // --- SELECTION: sync inbound selected key from parent/PieChart ---
        Set(varLegSelKey, Coalesce(Self.SelectedSliceKey, ""));

        // --- PADDING: cache padding inputs; +10 on right reserves space for scrollbar ---
        Set(varPadT, Coalesce(Self.PaddingTop, 4));
        Set(varPadB, Coalesce(Self.PaddingBottom, 4));
        Set(varPadL, Coalesce(Self.PaddingLeft, 4));
        Set(varPadR, Coalesce(Self.PaddingRight, 4));

        // --- ORIENTATION: vertical when taller than wide; horizontal enables multi-column wrap ---
        Set(varIsVertical, Self.Height >= Self.Width);

        // --- ROW HEIGHT: auto-fit rows to available height, clamped 18–28px ---
        Set(varItemCount, CountRows(Self.LegendData));
        Set(varAvailH, Self.Height - varPadT - varPadB);
        Set(
          varRowH,
          Max(
            18,
            Min(
              28,
              If(
                varItemCount > 0,
                RoundDown(varAvailH / varItemCount, 0),
                28
              )
            )
          )
        );

        // --- WRAP COUNT: 1 column when vertical; up to 4 columns when horizontal (130px each) ---
        Set(
          varWrapCount,
          If(
            varIsVertical,
            1,
            Min(4, Max(1, RoundDown((Self.Width - varPadL - varPadR) / 130, 0)))
          )
        )
      SelectedSliceKeyOut: =varLegSelKey
      SuggestedHeight: |-
        =// Recommended height: top padding + (rows * fixed row height) + bottom padding.
        // Uses a fixed _rowH of 24 (mid-point of the 18–28 clamp) for a stable suggestion.
        With(
          {
            _itemCount: CountRows(Self.LegendData),
            _padT: Coalesce(Self.PaddingTop, 4),
            _padB: Coalesce(Self.PaddingBottom, 4),
            _rowH: 24
          },
          _padT + (_itemCount * _rowH) + _padB
        )
      SuggestedWidth: |-
        =// Recommended width: sum of all visible column widths + gaps + padding.
        // _padR includes +10 to match the scrollbar reservation applied in OnReset.
        // Column widths: swatch=20, label=120, percent=52, value=58; gap=4 between each.
        With(
          {
            _padL: Coalesce(Self.PaddingLeft, 4),
            _padR: Coalesce(Self.PaddingRight, 4) + 10,
            _swatchW: If(Coalesce(Self.ShowColorBlock, true), 20, 0),
            _labelW: 120,
            _pctW: If(Coalesce(Self.ShowPercent, true), 52, 0),
            _valW: If(Coalesce(Self.ShowValue, false), 58, 0),
            _gap: 4
          },
          _padL + _swatchW + _gap + _labelW + _gap + _pctW + _gap + _valW + _padR
        )
      Width: =260
    Children:

                Coalesce(cmpPieChartLegend.ShadowSize, "none") = "border",

            RadiusBottomRight: =6

            RadiusTopRight: =6
