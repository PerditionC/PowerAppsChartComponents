    Properties:
      Height: =400
      Initialized: =varInitialized
      OnReset: |-
        =Set(varInitialized,true);

        // ====
        // SECTION 0: SYNC INBOUND SELECTION
        // ====
        // If the parent PieChart pushes a selection key, adopt it.
        // This runs on reset so the legend reflects the chart state.

        Set(
          varLegSelKey,
          Coalesce(Self.SelectedSliceKey, "")
        );

        // ====
        // SECTION 1: LAYOUT MODE
        // ====
        // Determine whether to use vertical or horizontal gallery layout.
        // Vertical when the component is taller than wide (legend beside pie).
        // Horizontal (multi-column wrap) when wider than tall (legend below pie).

        Set(varPadT, Coalesce(Self.PaddingTop, 4));
        Set(varPadB, Coalesce(Self.PaddingBottom, 4));
        Set(varPadL, Coalesce(Self.PaddingLeft, 4));
        Set(varPadR, Coalesce(Self.PaddingRight, 4));
        Set(varIsVertical, Self.Height >= Self.Width);
        Set(varRowH, 22);

        // ====
        // SECTION 2: COLUMN COUNT FOR HORIZONTAL MODE
        // ====
        // When laid out horizontally (wider than tall), use multi-column
        // wrapping so the legend does not waste vertical space.
        // Each column needs ~130px; clamp to [1, 4].

        Set(
          varWrapCount,
          If(
            varIsVertical,
            1,
            Min(4, Max(1, RoundDown((Self.Width - varPadL - varPadR) / 130, 0)))
          )
        );
      SelectedSliceKeyOut: =varLegSelKey
      SuggestedHeight: "=With(\r\n  {\r\n    _itemCount: CountRows(Self.LegendData),\r\n    _padT: Coalesce(Self.PaddingTop, 6),\r\n    _padB: Coalesce(Self.PaddingBottom, 6),\r\n    _itemH: 22  // Fixed item height\r\n  },\r\n  \r\n  If(\r\n    // VERTICAL ORIENTATION (tall, narrow)\r\n    Coalesce(Self.Orientation, Layout.Vertical) = Layout.Vertical,\r\n    \r\n    _padT + (_itemCount * _itemH) + _padB,\r\n    \r\n    // HORIZONTAL ORIENTATION (wide, short)\r\n    With(\r\n      {\r\n        // Calculate rows needed (assume 4 columns max)\r\n        _cols: Min(4, Max(1, _itemCount))\r\n      },\r\n    With(\r\n      {\r\n        _rows: RoundUp(_itemCount / _cols, 0)\r\n      },\r\n      \r\n      _padT + (_rows * _itemH) + _padB\r\n    )\r\n    )\r\n  )\r\n)"
      SuggestedWidth: "=With(\r\n  {\r\n    _itemCount: CountRows(Self.LegendData),\r\n    _padL: Coalesce(Self.PaddingLeft, 6),\r\n    _padR: Coalesce(Self.PaddingRight, 6),\r\n    _showBlock: Coalesce(Self.ShowColorBlock, true),\r\n    _showLabel: Coalesce(Self.ShowLabel, true),\r\n    _showValue: Coalesce(Self.ShowValue, true),\r\n    _itemH: 22,  // Fixed item height\r\n    _blockW: 14, // Color block width\r\n    _spacing: 4  // Spacing between elements\r\n  },\r\n  \r\n  If(\r\n    // VERTICAL ORIENTATION (tall, narrow)\r\n    Coalesce(Self.Orientation, Layout.Vertical) = Layout.Vertical,\r\n\r\n    With(\r\n      {\r\n        _labelLens: Sort(ForAll(\r\n                ShowColumns(Self.LegendData,Label),\r\n                Len(Coalesce(Text(ThisRecord.Label), \"\")) * 9\r\n              ), Value, SortOrder.Descending)\r\n      },\r\n    With(\r\n      {\r\n        // Estimate max label width (7px per character, max 20 chars visible)\r\n        _maxLabelW: If(\r\n          _showLabel,\r\n          Min(\r\n            First(_labelLens).Value,\r\n            140  // Max label width cap\r\n          ),\r\n          0\r\n        ),\r\n        // Estimate value width (format: \"1,234\" = ~45px)\r\n        _maxValueW: If(_showValue, 50, 0)\r\n      },\r\n      \r\n      _padL +\r\n      If(_showBlock, _blockW + _spacing, 0) +\r\n      _maxLabelW +\r\n      If(_showLabel And _showValue, _spacing, 0) +\r\n      _maxValueW +\r\n      _padR\r\n    )\r\n    ),\r\n    \r\n    // HORIZONTAL ORIENTATION (wide, short)\r\n    With(\r\n      {\r\n        // Estimate item width for horizontal layout\r\n        _labelW: If(_showLabel, 80, 0),  // Average label width\r\n        _valueW: If(_showValue, 50, 0)\r\n      },\r\n      With(\r\n      {\r\n        _itemW: If(_showBlock, _blockW + _spacing, 0) + _labelW + \r\n                If(_showLabel And _showValue, _spacing, 0) + _valueW + 8,\r\n        // Assume 4 columns max for horizontal\r\n        _cols: Min(4, Max(1, _itemCount))\r\n      },\r\n      \r\n      _padL + (_cols * _itemW) + _padR\r\n    )\r\n    )\r\n  )\r\n)"
      Width: =200
