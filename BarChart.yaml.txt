        DisplayName: Caption Color
        Description: CSS color for the caption text
        DataType: Text
        Default: ="black"
      CategoryAxisTitle:
        PropertyKind: Input
        DisplayName: Category Axis Title
        Description: Optional title for the category axis
        DataType: Text
        Default: =""
      CategoryLabelAngle:
        PropertyKind: Input
        DisplayName: Category Label Angle
        Description: Rotation angle for category labels (0, 45, or 90 degrees)
        DataType: Number
        Default: =0
      ChartBackgroundColor:
        PropertyKind: Input
        DisplayName: Chart Background Color
        Description: Fill color for the container behind the chart
        DataType: Text
        Default: ="#FAFAFA"
      ChartBorderColor:
        PropertyKind: Input
        DisplayName: Chart Border Color
        Description: Border color when ChartShadow is None
        DataType: Text
        Default: ="#E0E0E0"
      ChartMode:
        PropertyKind: Input
        DisplayName: Chart Mode
        Description: Layout mode for bars. "Simple" = one bar per category, "Grouped" = bars side by side per category, "Stacked" = bars stacked per category, "Stacked100" = stacked normalized to 100 percent.
        DataType: Text

      ChartShadow:
        PropertyKind: Input
        DisplayName: Chart Shadow
        Description: Visual effect for chart container - shadow, border, or none
        DataType: Text
        Default: =DropShadow.Semibold
      ChartTitle:
        PropertyKind: Input
        DisplayName: Chart Title
        Description: Text displayed as the chart heading
        DataType: Text
        Default: ="Bar Chart Title"
      EmptyStateText:
        PropertyKind: Input
        DisplayName: Empty State Text
        Description: Message shown when Items is empty or total is zero
        DataType: Text
        Default: ="No data"
      GroupGapRatio:
        PropertyKind: Input
        DisplayName: Group Gap Ratio
        Description: Ratio of gap width to group width between category groups (0 to 0.8). Larger values give more space between groups.
        DataType: Number
        Default: =0.25
      HasSelection:
        PropertyKind: Output
        DisplayName: Has Selection
        Description: True when a bar segment is currently selected
        DataType: Boolean
      IsEmpty:
        PropertyKind: Output
        DisplayName: Is Empty
        Description: True when Items is empty or all values are zero
        DataType: Boolean
      IsPrinting:
        PropertyKind: Input
        DisplayName: IsPrinting
        Description: Set to true (e.g. Screen.Printing) to hide interactive overlays for PDF export
        DataType: Boolean
        Default: =false
      Items:
        PropertyKind: Input
        DisplayName: Items
        Description: 'Data table. Required columns: Order (Number), Category (Text), Series (Text), Value (Number). Optional columns: Color (Text CSS hex), DrawAs (Text: Bar or Line), MarkerType (Text: None, Circle, FilledCircle, Diamond, Triangle).'
        RaiseOnReset: true
        DataType: Table
        Default: |-
          =Table(
            { Order: 1, Category: "Q1", Series: "Product A", Value: 45, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 2, Category: "Q1", Series: "Product B", Value: 32, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 3, Category: "Q1", Series: "Product C", Value: 18, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 4, Category: "Q1", Series: "Trend", Value: 38, Color: "#EF4444", DrawAs: "Line", MarkerType: "FilledCircle" },
            { Order: 5, Category: "Q2", Series: "Product A", Value: 72, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 6, Category: "Q2", Series: "Product B", Value: 58, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 7, Category: "Q2", Series: "Product C", Value: 29, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 8, Category: "Q2", Series: "Trend", Value: 65, Color: "#EF4444", DrawAs: "Line", MarkerType: "FilledCircle" },
            { Order: 9, Category: "Q3", Series: "Product A", Value: 58, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 10, Category: "Q3", Series: "Product B", Value: 41, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 11, Category: "Q3", Series: "Product C", Value: 35, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 12, Category: "Q3", Series: "Trend", Value: 50, Color: "#EF4444", DrawAs: "Line", MarkerType: "FilledCircle" },
            { Order: 13, Category: "Q4", Series: "Product A", Value: 91, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 14, Category: "Q4", Series: "Product B", Value: 67, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 15, Category: "Q4", Series: "Product C", Value: 44, Color: "", DrawAs: "Bar", MarkerType: "" },
            { Order: 16, Category: "Q4", Series: "Trend", Value: 80, Color: "#EF4444", DrawAs: "Line", MarkerType: "FilledCircle" }
          )
      LegendData:
        PropertyKind: Output
        DisplayName: Legend Data
        Description: 'Table of legend records for external legend component. Fields: SerIdx (Number), SerName (Text), SerColor (Text), DrawAs (Text), IsSelected (Boolean).'
        DataType: Table
      MajorTickInterval:
        PropertyKind: Input
        DisplayName: Major Tick Interval
        Description: Interval between major tick marks on the value axis. Leave blank for auto.
        DataType: Number
        Default: =0
      MinorTickInterval:
        PropertyKind: Input
        DisplayName: Minor Tick Interval
        Description: Interval between minor tick marks on the value axis. Leave blank for auto.
        DataType: Number
        Default: =0
      OnBarSelect:
        PropertyKind: Event
        DisplayName: On Bar Select
        Description: Fires when a bar segment or line point is selected
        ReturnType: None
        Default: =
      Orientation:
        PropertyKind: Input
        DisplayName: Orientation
        Description: '"Vertical" for vertical bars (categories on X axis), "Horizontal" for horizontal bars (categories on Y axis).'
        DataType: Text
        Default: ="Vertical"
      PaddingBottom:
        PropertyKind: Input
        DisplayName: Padding Bottom
        Description: Space between bottom component edge and content
        DataType: Number
        Default: =6
      PaddingLeft:
        PropertyKind: Input
        DisplayName: Padding Left
        Description: Space between left component edge and content
        DataType: Number
        Default: =6
      PaddingRight:
        PropertyKind: Input
        DisplayName: Padding Right
        Description: Space between right component edge and content
        DataType: Number
        Default: =6
      PaddingTop:
        PropertyKind: Input
        DisplayName: Padding Top
        Description: Space between top component edge and content
        DataType: Number
        Default: =6
      Palette:
        PropertyKind: Input
        DisplayName: Palette
        Description: Table of color strings assigned to series when no Color value is specified
        DataType: Table
        Default: |-
          =Table(
            { Value: "#2563EB" },
            { Value: "#10B981" },
            { Value: "#F59E0B" },
            { Value: "#EF4444" },
            { Value: "#8B5CF6" },
            { Value: "#EC4899" },
            { Value: "#14B8A6" },
            { Value: "#F97316" }
          )
      SelectedBarKey:
        PropertyKind: Input
        DisplayName: Selected Bar Key
        Description: Set this to a segment key (Category|Series) to select it externally (e.g. from a legend). Works alongside internal button selection.
        RaiseOnReset: true
        DataType: Text
        Default: =""
      SelectedCategory:
        PropertyKind: Output
        DisplayName: Selected Category
        Description: The Category of the currently selected segment
        DataType: Text
      SelectedKey:
        PropertyKind: Output
        DisplayName: Selected Key
        Description: The composite key (Category|Series) of the currently selected segment
        DataType: Text
      SelectedSegment:
        PropertyKind: Output
        DisplayName: Selected Segment
        Description: Full record of the currently selected segment from Items
        DataType: Record
      SelectedSeries:
        PropertyKind: Output
        DisplayName: Selected Series
        Description: The Series name of the currently selected segment
        DataType: Text
      ShowGridLines:
        PropertyKind: Input
        DisplayName: Show Grid Lines
        Description: Display grid lines across the plot area at major tick values
        DataType: Boolean
        Default: =true
      ShowValueLabels:
        PropertyKind: Input
        DisplayName: Show Value Labels
        Description: Display the numeric value on or above each bar segment
        DataType: Boolean
        Default: =false
      SvgMarkup:
        PropertyKind: Output
        DisplayName: SVG Markup
        Description: The raw SVG string used to render the chart (export-ready)
        DataType: Text
      ValueAxisMax:
        PropertyKind: Input
        DisplayName: Value Axis Max
        Description: Maximum value on the value axis. Leave blank (0) for auto.
        DataType: Number
        Default: =0
      ValueAxisMin:
        PropertyKind: Input
        DisplayName: Value Axis Min
        Description: Minimum value on the value axis. Leave blank (0) for auto.
        DataType: Number
        Default: =0
      ValueAxisTitle:
        PropertyKind: Input
        DisplayName: Value Axis Title
        Description: Optional title for the value axis
        DataType: Text
        Default: =""
    Properties:
      HasSelection: =varSelectedKey <> ""
      Height: =500
      IsEmpty: |-
        =IsEmpty(cmpBarChart.Items) Or
        Sum(cmpBarChart.Items, Coalesce(Value, 0)) = 0
      LegendData: |-
        =AddColumns(
          varSeriesList,
          IsSelected,
            SerName = LookUp(varBarSegments, SegKey = varSelectedKey).Series
        )
      OnReset: |-
        =// ====
        // SECTION 0: EXTERNAL SELECTION SYNC
        // ====
        Set(debugResetVisible, true);

        If(
          Coalesce(Self.SelectedBarKey, "") <> "",
          Set(varSelectedKey, Self.SelectedBarKey)
        );

        // ====
        // SECTION 1: PADDING AND TEXT HEIGHTS
        // ====
        Set(varPadT, Coalesce(Self.PaddingTop, 6));
        Set(varPadB, Coalesce(Self.PaddingBottom, 6));
        Set(varPadL, Coalesce(Self.PaddingLeft, 6));
        Set(varPadR, Coalesce(Self.PaddingRight, 6));
        Set(varTitleH, If(IsBlank(Self.ChartTitle), 0, 24));
        Set(varCaptionH, If(IsBlank(Self.Caption), 0, 30));

        // ====
        // SECTION 2: MODE, ORIENTATION, AND FLAGS
        // ====
        Set(varMode, Coalesce(Self.ChartMode, "Grouped"));
        Set(varIsHoriz, Lower(Coalesce(Self.Orientation, "Vertical")) = "horizontal");
        Set(varLabelAngle, Coalesce(Self.CategoryLabelAngle, 0));
        Set(varShowGrid, Coalesce(Self.ShowGridLines, true));
        Set(varShowValues, Coalesce(Self.ShowValueLabels, false));

        // Chart pixel dimensions (used for image control sizing)
        Set(varChartW, Self.Width - varPadL - varPadR);
        Set(varChartH, Self.Height - varPadT - varPadB - varTitleH - varCaptionH);

        // ====
        // SECTION 3: EXTRACT CATEGORIES AND SERIES
        // ====
        Set(palCount, CountRows(Self.Palette));

        // Ordered category list derived from Items.Order
        Set(varCategories,
          With(
            { _sorted: Sort(Self.Items, Order) },
            ForAll(
              Sequence(CountRows(Distinct(_sorted, Category))) As idx,
              {
                CatIdx: idx.Value,
                CatName: Last(FirstN(Distinct(_sorted, Category), idx.Value)).Value
              }
            )
          )
        );
        Set(varCatCount, CountRows(varCategories));

        // Ordered series list with resolved colors and draw modes
        Set(varSeriesList,
          With(
            { _sorted: Sort(Self.Items, Order) },
            ForAll(
              Sequence(CountRows(Distinct(_sorted, Series))) As idx,
              With(
                { _serName: Last(FirstN(Distinct(_sorted, Series), idx.Value)).Value },
                {
                  SerIdx: idx.Value,
                  SerName: _serName,
                  SerColor: Coalesce(
                    If(
                      Coalesce(LookUp(Self.Items, Series = _serName).Color, "") <> "",
                      LookUp(Self.Items, Series = _serName).Color,
                      Blank()
                    ),
                    Index(
                      Self.Palette,
                      Mod(idx.Value - 1, If(palCount = 0, 1, palCount)) + 1
                    ).Value,
                    "#D1D5DB"
                  ),
                  SerDrawAs: Lower(Coalesce(LookUp(Self.Items, Series = _serName).DrawAs, "bar")),
                  SerMarker: Coalesce(LookUp(Self.Items, Series = _serName).MarkerType, "None")
                }
              )
            )
          )
        );
        Set(varSerCount, CountRows(varSeriesList));
        Set(varBarSerCount, CountIf(varSeriesList, SerDrawAs = "bar"));

        // ====
        // SECTION 4: VALUE AXIS RANGE
        // ====
        // Max category total for stacked modes
        Set(varMaxCatTotal,
          Max(
            AddColumns(
              varCategories,
              _catTotal,
              Sum(
                Filter(Self.Items,
                  Category = CatName And
                  Lower(Coalesce(LookUp(varSeriesList, SerName = Series).SerDrawAs, "bar")) = "bar"
                ),
                Coalesce(Value, 0)
              )
            ),
            _catTotal
          )
        );

        Set(varYMin,
          If(
            varMode = "Stacked100", 0,
            If(
              Coalesce(Self.ValueAxisMin, 0) <> 0,
              Self.ValueAxisMin,
              Min(0, Min(Self.Items, Coalesce(Value, 0)))
            )
          )
        );

        Set(varAutoMax,
          If(
            varMode = "Stacked100", 100,
            If(
              Or(varMode = "Stacked", varMode = "Stacked100"),
              varMaxCatTotal,
              Max(Self.Items, Coalesce(Value, 0))
            )
          )
        );

        // Round up auto max to a nice number
        Set(varAutoMaxRounded,
          If(
            varAutoMax <= 0, 10,
            With(
              { _mag: Power(10, RoundDown(Log(varAutoMax, 10), 0)) },
              RoundUp(varAutoMax / _mag, 1) * _mag
            )
          )
        );

        Set(varYMax,
          If(
            varMode = "Stacked100", 100,
            If(
              Coalesce(Self.ValueAxisMax, 0) <> 0,
              Self.ValueAxisMax,
              varAutoMaxRounded
            )
          )
        );

        Set(varYRange, Max(varYMax - varYMin, 0.001));

        // Tick intervals (auto if zero/blank)
        Set(varMajorTick,
          If(
            Coalesce(Self.MajorTickInterval, 0) > 0,
            Self.MajorTickInterval,
            varYRange / 5
          )
        );
        Set(varMinorTick,
          If(
            Coalesce(Self.MinorTickInterval, 0) > 0,
            Self.MinorTickInterval,
            varMajorTick / 2
          )
        );

        // ====
        // SECTION 5: SVG VIEWBOX AND PLOT AREA
        // ====
        Set(vbW, 1000);
        Set(vbH, 700);

        // Margins in viewBox units -- accommodate labels and titles
        Set(varMLeft, If(varIsHoriz, 130, 75));
        Set(varMRight, 30);
        Set(varMTop, 20);
        Set(varMBottom, If(varIsHoriz, 50, If(varLabelAngle >= 90, 110, If(varLabelAngle >= 45, 90, 55))));

        Set(varPlotX, varMLeft);
        Set(varPlotY, varMTop);
        Set(varPlotW, vbW - varMLeft - varMRight);
        Set(varPlotH, vbH - varMTop - varMBottom);

        // ====
        // SECTION 6: BAR GEOMETRY PARAMETERS
        // ====
        Set(varGapRatio,
          Min(0.8, Max(0.05, Coalesce(Self.GroupGapRatio, 0.25)))
        );

        // Category axis length and value axis length in viewBox units
        Set(varCatAxisLen, If(varIsHoriz, varPlotH, varPlotW));
        Set(varValAxisLen, If(varIsHoriz, varPlotW, varPlotH));

        Set(varGroupW, varCatAxisLen / Max(varCatCount, 1));
        Set(varGapW, varGroupW * varGapRatio);
        Set(varUsableW, varGroupW - varGapW);

        // Individual bar width (grouped: divided among bar series; else full usable)
        Set(varBarW,
          If(
            varMode = "Grouped",
            varUsableW / Max(varBarSerCount, 1),
            varUsableW
          )
        );

        // ====
        // SECTION 7: BUILD varBarSegments
        // ====
        // One record per item with computed bar geometry in viewBox coords.
        Set(varBarSegments,
          ForAll(
            Self.Items,
            With(
              {
                _cat: LookUp(varCategories, CatName = Category),
                _ser: LookUp(varSeriesList, SerName = Series),
                _val: Coalesce(Value, 0)
              },
              With(
                {
                  _catIdx: Coalesce(_cat.CatIdx, 1),
                  _serIdx: Coalesce(_ser.SerIdx, 1),
                  _color: Coalesce(_ser.SerColor, "#D1D5DB"),
                  _drawAs: Coalesce(_ser.SerDrawAs, "bar"),
                  _marker: Coalesce(_ser.SerMarker, "None"),
                  _cumInCat: If(
                    Or(varMode = "Stacked", varMode = "Stacked100"),
                    Coalesce(
                      Sum(
                        Filter(Self.Items,
                          Category = ThisRecord.Category And
                          Coalesce(LookUp(varSeriesList, SerName = Series).SerIdx, 0) < Coalesce(_ser.SerIdx, 0) And
                          Lower(Coalesce(LookUp(varSeriesList, SerName = Series).SerDrawAs, "bar")) = "bar"
                        ),
                        Coalesce(Value, 0)
                      ), 0),
                    0
                  ),
                  _catTotal: Max(
                    Sum(Filter(Self.Items,
                      Category = ThisRecord.Category And
                      Lower(Coalesce(LookUp(varSeriesList, SerName = Series).SerDrawAs, "bar")) = "bar"
                    ), Coalesce(Value, 0)),
                    0.001
                  ),
                  _barPosIdx: If(
                    varMode = "Grouped",
                    CountIf(varSeriesList, SerIdx <= Coalesce(_ser.SerIdx, 0) And SerDrawAs = "bar"),
                    1
                  )
                },
                With(
                  {
                    _dispVal: If(varMode = "Stacked100", _val / _catTotal * 100, _val),
                    _cumDisp: If(varMode = "Stacked100", _cumInCat / _catTotal * 100, _cumInCat)
                  },
                  With(
                    {
                      _v0: If(
                        Or(varMode = "Stacked", varMode = "Stacked100"),
                        Max((_cumDisp - varYMin) / varYRange, 0),
                        Max((0 - varYMin) / varYRange, 0)
                      ),
                      _v1: If(
                        Or(varMode = "Stacked", varMode = "Stacked100"),
                        Max((_cumDisp + _dispVal - varYMin) / varYRange, 0),
                        Max((_dispVal - varYMin) / varYRange, 0)
                      ),
                      _catGroupStart: (_catIdx - 1) / Max(varCatCount, 1),
                      _catGroupMid: (_catIdx - 0.5) / Max(varCatCount, 1),
                      _gapFrac: varGapW / (2 * varCatAxisLen)
                    },
                    With(
                      {
                        _catBarStart: If(
                          varMode = "Grouped",
                          _catGroupStart + _gapFrac + (_barPosIdx - 1) * varBarW / varCatAxisLen,
                          _catGroupStart + _gapFrac
                        ),
                        _catBarW: If(varMode = "Grouped", varBarW, varUsableW) / varCatAxisLen
                      },
                      With(
                        {
                          _bx: If(varIsHoriz,
                            varPlotX + _v0 * varPlotW,
                            varPlotX + _catBarStart * varPlotW
                          ),
                          _by: If(varIsHoriz,
                            varPlotY + _catBarStart * varPlotH,
                            varPlotY + varPlotH * (1 - _v1)
                          ),
                          _bw: If(varIsHoriz,
                            Max((_v1 - _v0) * varPlotW, 0),
                            Max(_catBarW * varPlotW, 0)
                          ),
                          _bh: If(varIsHoriz,
                            Max(_catBarW * varPlotH, 0),
                            Max((_v1 - _v0) * varPlotH, 0)
                          ),
                          _px: If(varIsHoriz,
                            varPlotX + _v1 * varPlotW,
                            varPlotX + _catGroupMid * varPlotW
                          ),
                          _py: If(varIsHoriz,
                            varPlotY + _catGroupMid * varPlotH,
                            varPlotY + varPlotH * (1 - _v1)
                          )
                        },
                        {
                          SegKey: Category & "|" & Series,
                          Category: Category,
                          Series: Series,
                          Value: _val,
                          DisplayValue: _dispVal,
                          CatIdx: _catIdx,
                          SerIdx: _serIdx,
                          Color: _color,
                          DrawAs: _drawAs,
                          MarkerType: _marker,
                          BarX: _bx,
                          BarY: _by,
                          BarW: _bw,
                          BarH: _bh,
                          PointX: _px,
                          PointY: _py,
                          Tooltip: Category & " / " & Series & ": " & Text(_val)
                        }
                      )
                    )
                  )
                )
              )
            )
          )
        );

        // ====
        // SECTION 8: SVG ASSEMBLY
        // ====
        // Format shorthand used throughout
        // f = "[$-en-US]0.00"

        // 8a: Grid lines at each minor tick on the value axis
        // Major grid lines are drawn with a heavier stroke; minor with a lighter one.
        Set(varSvgGrid,
          If(Not(varShowGrid), "",
            Concat(
              Filter(
                ForAll(
                  Sequence(RoundDown(varYRange / Max(varMinorTick, 0.001), 0) + 1) As tN,
                  {
                    _tv: varYMin + (tN.Value - 1) * varMinorTick,
                    _isMaj: Abs(Mod(Round((tN.Value - 1) * varMinorTick, 6), Round(varMajorTick, 6))) < 0.001
                       Or (tN.Value - 1) * varMinorTick = 0
                  }
                ),
                _tv <= varYMax And _tv > varYMin
              ),
              With(
                {
                  _frac: (_tv - varYMin) / varYRange,
                  _sw: If(_isMaj, "1", "0.5"),
                  _sc: If(_isMaj, "#CBD5E1", "#E5E7EB")
                },
                If(varIsHoriz,
                  "<line x1='" & Text(varPlotX + _frac * varPlotW, "[$-en-US]0.00") &
                  "' y1='" & Text(varPlotY, "[$-en-US]0.00") &
                  "' x2='" & Text(varPlotX + _frac * varPlotW, "[$-en-US]0.00") &
                  "' y2='" & Text(varPlotY + varPlotH, "[$-en-US]0.00") &
                  "' stroke='" & _sc & "' stroke-width='" & _sw & "'/>",
                  "<line x1='" & Text(varPlotX, "[$-en-US]0.00") &
                  "' y1='" & Text(varPlotY + varPlotH * (1 - _frac), "[$-en-US]0.00") &
                  "' x2='" & Text(varPlotX + varPlotW, "[$-en-US]0.00") &
                  "' y2='" & Text(varPlotY + varPlotH * (1 - _frac), "[$-en-US]0.00") &
                  "' stroke='" & _sc & "' stroke-width='" & _sw & "'/>"
                )
              )
            )
          )
        );

        // 8b: Bar rectangles (only for bar-type series)
        Set(varSvgBars,
          Concat(
            Sort(Filter(varBarSegments, DrawAs = "bar"), CatIdx * 1000 + SerIdx),
            "<rect x='" & Text(BarX, "[$-en-US]0.00") &
            "' y='" & Text(BarY, "[$-en-US]0.00") &
            "' width='" & Text(Max(BarW, 0), "[$-en-US]0.00") &
            "' height='" & Text(Max(BarH, 0), "[$-en-US]0.00") &
            "' fill='" & Color &
            "' rx='2'/>"
          )
        );

        // 8c: Line series -- one polyline per line-type series
        Set(varSvgLines,
          Concat(
            Filter(varSeriesList, SerDrawAs <> "bar"),
            "<polyline points='" &
            Concat(
              Sort(Filter(varBarSegments, Series = SerName And DrawAs <> "bar"), CatIdx),
              Text(PointX, "[$-en-US]0.00") & "," & Text(PointY, "[$-en-US]0.00"),
              " "
            ) &
            "' fill='none' stroke='" & SerColor &
            "' stroke-width='2.5' stroke-linejoin='round'/>"
          )
        );

        // 8d: Line markers at each data point
        Set(varSvgMarkers,
          Concat(
            Filter(varBarSegments, DrawAs <> "bar"),
            With(
              {
                _mx: Text(PointX, "[$-en-US]0.00"),
                _my: Text(PointY, "[$-en-US]0.00"),
                _mc: Color,
                _mt: MarkerType
              },
              If(
                _mt = "Circle",
                "<circle cx='" & _mx & "' cy='" & _my & "' r='5' fill='white' stroke='" & _mc & "' stroke-width='2'/>",
                _mt = "FilledCircle",
                "<circle cx='" & _mx & "' cy='" & _my & "' r='5' fill='" & _mc & "' stroke='white' stroke-width='1.5'/>",
                _mt = "Diamond",
                "<polygon points='" &
                  _mx & "," & Text(PointY - 6, "[$-en-US]0.00") & " " &
                  Text(PointX + 6, "[$-en-US]0.00") & "," & _my & " " &
                  _mx & "," & Text(PointY + 6, "[$-en-US]0.00") & " " &
                  Text(PointX - 6, "[$-en-US]0.00") & "," & _my &
                "' fill='" & _mc & "' stroke='white' stroke-width='1.5'/>",
                _mt = "Triangle",
