      HitButtonBorderStyle:

        RaiseOnReset: true
        DataType: Text
        Default: =BorderStyle.None
      HitButtonBorderWidth:
        PropertyKind: Input
        DisplayName: Hit Button Border Width
        Description: Border thickness in pixels for the hit-test buttons. 0 = none.
        RaiseOnReset: true
        DataType: Number
        Default: =0
      InteractiveBorderColor:
        PropertyKind: Input
        DisplayName: Interactive Border Color
        Description: Border color for the interactive overlay container. Default is transparent.
        RaiseOnReset: true
        DataType: Text
        Default: =""
      InteractiveBorderStyle:
        PropertyKind: Input
        DisplayName: Interactive Border Style
        Description: Border style for the interactive overlay container. "None", "Solid", "Dashed", "Dotted".
        RaiseOnReset: true
        DataType: Text
        Default: =BorderStyle.None
      InteractiveBorderWidth:
        PropertyKind: Input
        DisplayName: Interactive Border Width
        Description: Border thickness in pixels for the interactive overlay container. 0 = none.
        RaiseOnReset: true
        DataType: Number
        Default: =0
      IsEmpty:
        PropertyKind: Output
        DisplayName: Is Empty
        Description: True when Items is empty or all values are zero
        DataType: Boolean
      IsPrinting:
        PropertyKind: Input
        DisplayName: Is Printing
        Description: Set to Screen.Printing to hide interactive overlays for PDF export
        RaiseOnReset: true
        DataType: Boolean
        Default: =false
      Items:
        PropertyKind: Input
        DisplayName: Items
        Description: >-
          Data table. Supported columns:

          Category or X  (Text)   -- category axis label. Both names accepted; Category takes priority. Value    or Y  (Number) -- numeric value. Both names accepted; Value takes priority. Series         (Text)   -- series name. Defaults to "Series 1" when omitted. Order          (Number) -- sort order for categories and series. Optional. Color          (Text)   -- CSS hex color override for this series. Optional. DrawAs         (Text)   -- "Bar" (default) or "Line". Optional. MarkerType     (Text)   -- "None","Circle","FilledCircle","Diamond","Triangle". Optional.

          Key is NOT a valid input column; it is always derived as Category & "|" & Series.

          Include both X/Category and Y/Value columns in the table schema (superset pattern) so Coalesce references never fail on missing fields.

          Limit: top 20 categories and top 12 series by Order are used; extras are dropped.
        RaiseOnReset: true
        DataType: Table
        Default: |-
          =Table(

            { Order: 3,  X: "Q1", Category: "Q1-longName", Y: 18, Value: 18, Series: "Product C", Color: "", DrawAs: "Bar",  MarkerType: "" },
            { Order: 4,  X: "Q1", Category: "Q1-longName", Y: 38, Value: 38, Series: "Trend",     Color: "#EF4444", DrawAs: "Line", MarkerType: "FilledCircle" },
            { Order: 5,  X: "Q2", Category: "Q2-longName", Y: 72, Value: 72, Series: "Product A", Color: "", DrawAs: "Bar",  MarkerType: "" },
            { Order: 6,  X: "Q2", Category: "Q2-longName", Y: 58, Value: 58, Series: "Product B", Color: "", DrawAs: "Bar",  MarkerType: "" },
            { Order: 7,  X: "Q2", Category: "Q2-longName", Y: 29, Value: 29, Series: "Product C", Color: "", DrawAs: "Bar",  MarkerType: "" },
            { Order: 8,  X: "Q2", Category: "Q2-longName", Y: 65, Value: 65, Series: "Trend",     Color: "#EF4444", DrawAs: "Line", MarkerType: "Diamond" },
            { Order: 9,  X: "Q3", Category: "Q3-longName", Y: 58, Value: 58, Series: "Product A", Color: "", DrawAs: "Bar",  MarkerType: "" },
            { Order: 10, X: "Q3", Category: "Q3-longName", Y: 41, Value: 41, Series: "Product B", Color: "", DrawAs: "Bar",  MarkerType: "" },
            { Order: 11, X: "Q3", Category: "Q3-longName", Y: 35, Value: 35, Series: "Product C", Color: "", DrawAs: "Bar",  MarkerType: "" },
            { Order: 12, X: "Q3", Category: "Q3-longName", Y: 50, Value: 50, Series: "Trend",     Color: "#EF4444", DrawAs: "Line", MarkerType: "FilledCircle" },
            { Order: 13, X: "Q4", Category: "Q4", Y: 91, Value: 91, Series: "Product A", Color: "", DrawAs: "Bar",  MarkerType: "" },
            { Order: 14, X: "Q4", Category: "Q4", Y: 67, Value: 67, Series: "Product B", Color: "", DrawAs: "Bar",  MarkerType: "" },
            { Order: 15, X: "Q4", Category: "Q4", Y: 44, Value: 44, Series: "Product C", Color: "", DrawAs: "Bar",  MarkerType: "" },
            { Order: 16, X: "Q4", Category: "Q4", Y: 80, Value: 80, Series: "Trend",     Color: "#EF4444", DrawAs: "Line", MarkerType: "FilledCircle" }
          )
      LegendData:
        PropertyKind: Output
        DisplayName: Legend Data
        Description: 'Table for an external legend component. Fields: SerIdx, SerName, SerColor, DrawAs, IsSelected.'
        DataType: Table
      MajorTickInterval:
        PropertyKind: Input
        DisplayName: Major Tick Interval
        Description: Interval between major tick marks on the value axis. 0 = auto.
        RaiseOnReset: true
        DataType: Number
        Default: =0
      MinorTickInterval:
        PropertyKind: Input
        DisplayName: Minor Tick Interval
        Description: Interval between minor tick marks on the value axis. 0 = auto.
        RaiseOnReset: true
        DataType: Number
        Default: =0
      OnBarSelect:
        PropertyKind: Event
        DisplayName: On Bar Select
        Description: Fires when a bar segment or line point is tapped/clicked
        ReturnType: None
        Default: =
      Orientation:
        PropertyKind: Input
        DisplayName: Orientation
        Description: '"Vertical" = categories on X axis. "Horizontal" = categories on Y axis.'
        RaiseOnReset: true
        DataType: Text
        Default: ="Vertical" // "Horizontal"
      PaddingBottom:
        PropertyKind: Input
        DisplayName: Padding Bottom
        Description: Space between bottom component edge and content
        RaiseOnReset: true
        DataType: Number
        Default: =6
      PaddingLeft:
        PropertyKind: Input
        DisplayName: Padding Left
        Description: Space between left component edge and content
        RaiseOnReset: true
        DataType: Number
        Default: =6
      PaddingRight:
        PropertyKind: Input
        DisplayName: Padding Right
        Description: Space between right component edge and content
        RaiseOnReset: true
        DataType: Number
        Default: =6

        PropertyKind: Input
        DisplayName: Padding Top
        Description: Space between top component edge and content
        RaiseOnReset: true
        DataType: Number
        Default: =6
      Palette:
        PropertyKind: Input
        DisplayName: Palette
        Description: Table of CSS hex color strings cycled when a series has no Color value
        RaiseOnReset: true
        DataType: Table
        Default: |-
          =Table(
            { Value: "#2563EB" },
            { Value: "#10B981" },
            { Value: "#F59E0B" },
            { Value: "#EF4444" },
            { Value: "#8B5CF6" },
            { Value: "#EC4899" },
            { Value: "#14B8A6" },
            { Value: "#F97316" }
          )
      SelectedBarKey:
        PropertyKind: Input
        DisplayName: Selected Bar Key
        Description: Set externally (e.g. from a legend) to select a segment by its Key (Category & "|" & Series). Synced into varSelectedKey on reset.
        RaiseOnReset: true
        DataType: Text
        Default: =""
      SelectedCategory:
        PropertyKind: Output
        DisplayName: Selected Category
        Description: Category of the currently selected segment
        DataType: Text
      SelectedKey:
        PropertyKind: Output
        DisplayName: Selected Key
        Description: Composite key (Category & "|" & Series) of the selected segment
        DataType: Text
      SelectedSegment:
        PropertyKind: Output
        DisplayName: Selected Segment
        Description: Full normalized record of the currently selected segment
        DataType: Record
      SelectedSeries:
        PropertyKind: Output
        DisplayName: Selected Series
        Description: Series name of the currently selected segment
        DataType: Text
      ShowGridLines:
        PropertyKind: Input
        DisplayName: Show Grid Lines
        Description: Display grid lines across the plot area at major tick values
        RaiseOnReset: true
        DataType: Boolean
        Default: =true
      ShowValueLabels:
        PropertyKind: Input
        DisplayName: Show Value Labels
        Description: Display the numeric value on or above each bar segment
        RaiseOnReset: true
        DataType: Boolean
        Default: =true
      SvgMarkup:
        PropertyKind: Output
        DisplayName: SVG Markup
        Description: Raw SVG string used to render the chart (export-ready)
        DataType: Text
      ValueAxisMax:
        PropertyKind: Input
        DisplayName: Value Axis Max
        Description: Maximum value on the value axis. 0 = auto.
        RaiseOnReset: true
        DataType: Number
        Default: =0
      ValueAxisMin:
        PropertyKind: Input
        DisplayName: Value Axis Min
        Description: Minimum value on the value axis. 0 = auto (uses data min, floored at 0).
        RaiseOnReset: true
        DataType: Number
        Default: =0
      ValueAxisTitle:
        PropertyKind: Input
        DisplayName: Value Axis Title
        Description: Optional title for the value axis
        RaiseOnReset: true
        DataType: Text
        Default: ="Value title (Y axis)"
    Properties:
      HasSelection: =varSelectedKey <> ""
      Height: =500
      IsEmpty: |-
        =IsEmpty(varItemsNorm) Or
        Sum(varItemsNorm, Coalesce(Value, 0)) = 0
      LegendData: |-
        =AddColumns(
          varSeriesList,
          IsSelected,
            SerName = LookUp(varBarSegments, SegKey = varSelectedKey).Series
        )
      OnReset: |-
        =// Dev helper -- set false before shipping
        Set(debugResetVisible, true);

        // ====
        // SECTION 0: EXTERNAL SELECTION SYNC
        // ====
        If(
          Coalesce(cmpBarChart.SelectedBarKey, "") <> "",
          Set(varSelectedKey, cmpBarChart.SelectedBarKey)
        );

        // ====
        // SECTION 1: PADDING AND TEXT HEIGHTS
        // ====
        Set(varPadT,     Coalesce(cmpBarChart.PaddingTop,    6));
        Set(varPadB,     Coalesce(cmpBarChart.PaddingBottom, 6));
        Set(varPadL,     Coalesce(cmpBarChart.PaddingLeft,   6));
        Set(varPadR,     Coalesce(cmpBarChart.PaddingRight,  6));
        Set(varTitleH,   If(IsBlank(cmpBarChart.ChartTitle), 0, 24));
        Set(varCaptionH, If(IsBlank(cmpBarChart.Caption),    0, 30));

        Set(varChartW, cmpBarChart.Width  - varPadL - varPadR);
        Set(varChartH, cmpBarChart.Height - varTitleH - varCaptionH - varPadT - varPadB);

        // ====
        // SECTION 2: MODE, ORIENTATION, FLAGS
        // ====
        Set(varMode,       Coalesce(cmpBarChart.ChartMode, "Grouped"));
        Set(varIsHoriz,    Lower(Coalesce(cmpBarChart.Orientation, "Vertical")) = "horizontal");
        Set(varLabelAngle, Max(-90, Min(90, Coalesce(cmpBarChart.CategoryLabelAngle, 0))));
        Set(varShowGrid,   Coalesce(cmpBarChart.ShowGridLines,   true));
        Set(varShowValues, Coalesce(cmpBarChart.ShowValueLabels, false));
        Set(palCount,      CountRows(cmpBarChart.Palette));

        // ====
        // SECTION 3: INPUT NORMALIZATION
        // ====
        Set(
          varItemsNorm,
          ForAll(
            FirstN(
              SortByColumns(cmpBarChart.Items, "Order", SortOrder.Ascending),
              240
            ) As _r,
            {
              Order:      Coalesce(_r.Order, 0),
              Category:   Coalesce(Text(_r.Category), Text(_r.X), "?"),
              Value:      Coalesce(_r.Value, _r.Y, 0),
              Series:     Coalesce(Text(_r.Series), "Series 1"),
              Color:      Coalesce(Text(_r.Color), ""),
              DrawAs:     Lower(Coalesce(Text(_r.DrawAs), "bar")),
              MarkerType: Coalesce(Text(_r.MarkerType), "None")
            }
          )
        );

        // ====
        // SECTION 4: SERIES LIST, SERIES THINNING, CATEGORY LIST
        // ====

        // 4a. Raw series list
        Set(
          varSeriesList,
          FirstN(
            ForAll(
              Sequence(CountRows(Distinct(varItemsNorm, Series))) As idx,
              With(
                {
                  _sn: Last(
                    FirstN(
                      Distinct(
                        SortByColumns(varItemsNorm, "Order", SortOrder.Ascending),
                        Series
                      ),
                      idx.Value
                    )
                  ).Value
                },
                With(
                  { _row: LookUp(varItemsNorm, Series = _sn) },
                  {
                    SerIdx:    idx.Value,
                    SerName:   _sn,
                    SerColor:  Coalesce(
                      If(Coalesce(_row.Color, "") <> "", _row.Color, Blank()),
                      If(
                        palCount > 0,
                        Index(cmpBarChart.Palette, Mod(idx.Value - 1, palCount) + 1).Value,
                        Blank()
                      ),
                      "#D1D5DB"
                    ),
                    SerDrawAs: Coalesce(_row.DrawAs, "bar"),
                    SerMarker: Coalesce(_row.MarkerType, "None")
                  }
                )
              )
            ),
            12
          )
        );

        Set(varSerCount,    CountRows(varSeriesList));
        Set(varBarSerCount, CountIf(varSeriesList, SerDrawAs = "bar"));

        // 4b. Grouped-only thinning when >10 bar series
        Set(varMaxBarSeries, 10);

        If(
          varMode = "Grouped" And varBarSerCount > varMaxBarSeries,

          Set(
            varItemsNorm,
            ForAll(
              varItemsNorm As _r,
              With(
                { _si: LookUp(varSeriesList, SerName = _r.Series) },
                {
                  Order:      _r.Order,
                  Category:   _r.Category,
                  Value:      _r.Value,
                  Series:     If(
                    _si.SerDrawAs = "bar" And _si.SerIdx >= varMaxBarSeries,
                    "Other",
                    _r.Series
                  ),
                  Color:      _r.Color,
                  DrawAs:     _r.DrawAs,
                  MarkerType: _r.MarkerType
                }
              )
            )
          );

          Set(
            varSeriesList,
            FirstN(
              ForAll(
                Sequence(CountRows(Distinct(varItemsNorm, Series))) As idx,
                With(
                  {
                    _sn: Last(
                      FirstN(
                        Distinct(
                          SortByColumns(varItemsNorm, "Order", SortOrder.Ascending),
                          Series
                        ),
                        idx.Value
                      )
                    ).Value
                  },
                  With(
                    { _row: LookUp(varItemsNorm, Series = _sn) },
                    {
                      SerIdx:    idx.Value,
                      SerName:   _sn,
                      SerColor:  Coalesce(
                        If(Coalesce(_row.Color, "") <> "", _row.Color, Blank()),
                        If(
                          palCount > 0,
                          Index(cmpBarChart.Palette, Mod(idx.Value - 1, palCount) + 1).Value,
                          Blank()
                        ),
                        "#D1D5DB"
                      ),
                      SerDrawAs: Coalesce(_row.DrawAs, "bar"),
                      SerMarker: Coalesce(_row.MarkerType, "None")
                    }
                  )
                )
              ),
              12
            )
          );

          Set(varSerCount,    CountRows(varSeriesList));
          Set(varBarSerCount, CountIf(varSeriesList, SerDrawAs = "bar"))
        );

        // 4c. Simple mode keeps first bar series + all line series
        If(
          varMode = "Simple",
          Set(
            varItemsNorm,
            Filter(
              varItemsNorm,
              DrawAs <> "bar" Or
              LookUp(varSeriesList, SerName = Series).SerIdx =
                Min(Filter(varSeriesList, SerDrawAs = "bar"), SerIdx)
            )
          )
        );

        // 4d. Rebuild visible series list after any mode-specific filtering
        Set(
          varSeriesList,
          FirstN(
            ForAll(
              Sequence(CountRows(Distinct(varItemsNorm, Series))) As idx,
              With(
                {
                  _sn: Last(
                    FirstN(
                      Distinct(
                        SortByColumns(varItemsNorm, "Order", SortOrder.Ascending),
                        Series
                      ),
                      idx.Value
                    )
                  ).Value
                },
                With(
                  { _row: LookUp(varItemsNorm, Series = _sn) },
                  {
                    SerIdx:    idx.Value,
                    SerName:   _sn,
                    SerColor:  Coalesce(
                      If(Coalesce(_row.Color, "") <> "", _row.Color, Blank()),
                      If(
                        palCount > 0,
                        Index(cmpBarChart.Palette, Mod(idx.Value - 1, palCount) + 1).Value,
                        Blank()
                      ),
                      "#D1D5DB"
                    ),
                    SerDrawAs: Coalesce(_row.DrawAs, "bar"),
                    SerMarker: Coalesce(_row.MarkerType, "None")
                  }
                )
              )
            ),
            12
          )
        );

        Set(varSerCount,    CountRows(varSeriesList));
        Set(varBarSerCount, CountIf(varSeriesList, SerDrawAs = "bar"));

        // 4e. Categories
        Set(
          varCategories,
          FirstN(
            ForAll(
              Sequence(CountRows(Distinct(varItemsNorm, Category))) As idx,
              {
                CatIdx:  idx.Value,
                CatName: Last(
                  FirstN(
                    Distinct(
                      SortByColumns(varItemsNorm, "Order", SortOrder.Ascending),
                      Category
                    ),
                    idx.Value
                  )
                ).Value
              }
            ),
            20
          )
        );
        Set(varCatCount, CountRows(varCategories));

        // ====
        // SECTION 5: VALUE AXIS RANGE AND TICK INTERVALS
        // ====
        Set(
          varMaxCatTotal,
          Max(
            AddColumns(
              varCategories,
              _ct,
              Sum(
                Filter(
                  varItemsNorm,
                  Category = CatName And
                  Coalesce(LookUp(varSeriesList, SerName = Series).SerDrawAs, "bar") = "bar"
                ),
                Coalesce(Value, 0)
              )
            ),
            _ct
          )
        );

        Set(
          varYMin,
          If(
            varMode = "Stacked100",
            0,
            If(
              Coalesce(cmpBarChart.ValueAxisMin, 0) <> 0,
              cmpBarChart.ValueAxisMin,
              Min(0, Min(varItemsNorm, Coalesce(Value, 0)))
            )
          )
        );

        Set(
          varAutoMax,
          If(
            varMode = "Stacked100",
            100,
            If(
              Or(varMode = "Stacked", varMode = "Stacked100"),
              varMaxCatTotal,
              Max(varItemsNorm, Coalesce(Value, 0))
            )
          )
        );

        Set(
          varAutoMaxRounded,
          If(
            varAutoMax <= 0,
            10,
            With(
              { _mag: Power(10, RoundDown(Log(varAutoMax, 10), 0)) },
              RoundUp(varAutoMax / _mag, 1) * _mag
            )
          )
        );

        Set(
          varYMax,
          If(
            varMode = "Stacked100",
            100,
            If(
              Coalesce(cmpBarChart.ValueAxisMax, 0) <> 0,
              cmpBarChart.ValueAxisMax,
              varAutoMaxRounded
            )
          )
        );

        Set(varYRange, Max(varYMax - varYMin, 0.001));

        Set(
          varMajorTick,
          If(
            Coalesce(cmpBarChart.MajorTickInterval, 0) > 0,
            cmpBarChart.MajorTickInterval,
            varYRange / 5
          )
        );
        Set(
          varMinorTick,
          If(
            Coalesce(cmpBarChart.MinorTickInterval, 0) > 0,
            cmpBarChart.MinorTickInterval,
            varMajorTick / 2
          )
        );

        Set(varSafeMinorTick, Max(varMinorTick, 0.001));
        Set(varTickCount,     RoundDown(varYRange / varSafeMinorTick, 0) + 1);

        Set(
          varTickRows,
          ForAll(
            Sequence(varTickCount) As tN,
            {
              _tv:    varYMin + (tN.Value - 1) * varMinorTick,
              _isMaj: Abs(Mod(Round((tN.Value - 1) * varMinorTick, 6), Round(varMajorTick, 6))) < 0.001
                   Or (tN.Value - 1) * varMinorTick = 0
            }
          )
        );

        // ====
        // SECTION 6: SVG VIEWBOX AND PLOT RECTANGLE
        // ====
        Set(vbW, 1000);
        Set(vbH, RoundDown(1000 * varChartH / varChartW, 0));

        Set(varMLeft,  If(varIsHoriz, 130, 75));
        Set(varMRight, 30);
        Set(varMTop,   20);
        Set(
          varMBottom,
          If(
            varIsHoriz,
            50,
            If(
              Abs(varLabelAngle) >= 80,
              120,
              If(
                Abs(varLabelAngle) >= 45,
                92,
                If(Abs(varLabelAngle) >= 20, 72, 55)
              )
            )
          )
        );

        Set(varScaleX, varChartW / vbW);
        Set(varScaleY, varChartH / vbH);

        Set(varPlotX, varMLeft);
        Set(varPlotY, varMTop);
        Set(varPlotW, vbW - varMLeft - varMRight);
        Set(varPlotH, vbH - varMTop - varMBottom);

        Set(varPlotPixX, varPlotX * varScaleX);
        Set(varPlotPixY, varPlotY * varScaleY);
        Set(varPlotPixW, varPlotW * varScaleX);
        Set(varPlotPixH, varPlotH * varScaleY);

        // ====
        // SECTION 7: BAR GEOMETRY PARAMETERS
        // ====
        Set(varGapRatio, Min(0.8, Max(0.05, Coalesce(cmpBarChart.GroupGapRatio, 0.25))));

        Set(varCatAxisLen, If(varIsHoriz, varPlotH, varPlotW));
        Set(varValAxisLen, If(varIsHoriz, varPlotW, varPlotH));

        Set(varGroupW,  varCatAxisLen / Max(varCatCount, 1));
        Set(varGapW,    varGroupW * varGapRatio);
        Set(varUsableW, varGroupW - varGapW);

        Set(
          varBarW,
          If(
            varMode = "Grouped",
            varUsableW / Max(varBarSerCount, 1),
            varUsableW
          )
        );

        // ====
        // SECTION 8: BUILD varBarSegments
        // ====
        // Key fixes:
        // - Simple bars color by category.
        // - Stacked100 bars normalize for geometry, but labels keep raw values.
        // - Line series always plot at their own value, never at cumulative stacked height.
        Set(
          varBarSegments,
          ForAll(
            varItemsNorm As _outerRow,
            With(
              {
                _thisCat:    Coalesce(Text(_outerRow.Category), "?"),
                _thisSeries: Coalesce(Text(_outerRow.Series), "Series 1"),
                _thisVal:    Coalesce(_outerRow.Value, 0),
                _thisDrawAs: Lower(Coalesce(Text(_outerRow.DrawAs), "bar"))
              },
              With(
                {
                  _cat: LookUp(varCategories, CatName = _thisCat),
                  _ser: LookUp(varSeriesList, SerName = _thisSeries)
                },
                With(
                  {
                    _catIdx:    Coalesce(_cat.CatIdx, 1),
                    _serIdx:    Coalesce(_ser.SerIdx, 1),
                    _drawAs:    Coalesce(_ser.SerDrawAs, _thisDrawAs, "bar"),
                    _marker:    Coalesce(_outerRow.MarkerType, _ser.SerMarker, "None"),
                    _isBar:     Coalesce(_ser.SerDrawAs, _thisDrawAs, "bar") = "bar",

                    _color: If(
                      Coalesce(_ser.SerDrawAs, _thisDrawAs, "bar") = "bar" And (varMode = "Simple" Or varBarSerCount <= 1),
                      Coalesce(
                        If(
                          palCount > 0,
                          Index(cmpBarChart.Palette, Mod(Coalesce(_cat.CatIdx, 1) - 1, palCount) + 1).Value,
                          Blank()
                        ),
                        "#D1D5DB"
                      ),
                      Coalesce(_ser.SerColor, "#D1D5DB")
                    ),

                    _barPosIdx: If(
                      varMode = "Grouped",
                      CountIf(
                        varSeriesList,
                        SerDrawAs = "bar" And SerIdx <= Coalesce(_ser.SerIdx, 0)
                      ),
                      1
                    )
                  },
                  With(
                    {
                      _cumInCat: If(
                        _isBar And Or(varMode = "Stacked", varMode = "Stacked100"),
                        Coalesce(
                          Sum(
                            Filter(
                              varItemsNorm As _cand,
                              _cand.Category = _thisCat And
                              Lower(Coalesce(Text(_cand.DrawAs), "bar")) = "bar" And
                              Coalesce(LookUp(varSeriesList, SerName = _cand.Series).SerIdx, 0) < _serIdx
                            ),
                            Coalesce(ThisRecord.Value, 0)
                          ),
                          0
                        ),
                        0
                      ),

                      _catTotal: Max(
                        Sum(
                          Filter(
                            varItemsNorm As _cand,
                            _cand.Category = _thisCat And
                            Lower(Coalesce(Text(_cand.DrawAs), "bar")) = "bar"
                          ),
                          Coalesce(ThisRecord.Value, 0)
                        ),
                        0.001
                      )
                    },
                    With(
                      {
                        _barDispVal: If(
                          _isBar And varMode = "Stacked100",
                          _thisVal / _catTotal * 100,
                          _thisVal
                        ),
                        _cumDisp: If(
                          _isBar And varMode = "Stacked100",
                          _cumInCat / _catTotal * 100,
                          _cumInCat
                        ),
                        _lineDispVal: _thisVal
                      },
                      With(
                        {
                          _v0: If(
                            _isBar And Or(varMode = "Stacked", varMode = "Stacked100"),
                            Min(1, Max((_cumDisp - varYMin) / varYRange, 0)),
                            Min(1, Max((0 - varYMin) / varYRange, 0))
                          ),

                          _v1: If(
                            _isBar And Or(varMode = "Stacked", varMode = "Stacked100"),
                            Min(1, Max((_cumDisp + _barDispVal - varYMin) / varYRange, 0)),
                            If(
                              _isBar,
                              Min(1, Max((_barDispVal - varYMin) / varYRange, 0)),
                              Min(1, Max((_lineDispVal - varYMin) / varYRange, 0))
                            )
                          ),

                          _pointFrac: If(
                            _isBar,
                            If(
                              Or(varMode = "Stacked", varMode = "Stacked100"),
                              Min(1, Max((_cumDisp + _barDispVal - varYMin) / varYRange, 0)),
                              Min(1, Max((_barDispVal - varYMin) / varYRange, 0))
                            ),
                            Min(1, Max((_lineDispVal - varYMin) / varYRange, 0))
                          ),

                          _catGroupStart: (_catIdx - 1) / Max(varCatCount, 1),
                          _catGroupMid:   (_catIdx - 0.5) / Max(varCatCount, 1),
                          _gapFrac:       varGapW / (2 * varCatAxisLen)
                        },
                        With(
                          {
                            _catBarStart: If(
                              varMode = "Grouped",
                              _catGroupStart + _gapFrac + (_barPosIdx - 1) * varBarW / varCatAxisLen,
                              _catGroupStart + _gapFrac
                            ),
                            _catBarFrac: If(varMode = "Grouped", varBarW, varUsableW) / varCatAxisLen
                          },
                          With(
                            {
                              _bx: If(
                                varIsHoriz,
                                varPlotX + _v0 * varPlotW,
                                varPlotX + _catBarStart * varPlotW
                              ),
                              _by: If(
                                varIsHoriz,
                                varPlotY + _catBarStart * varPlotH,
                                varPlotY + varPlotH * (1 - _v1)
                              ),
                              _bw: If(
                                varIsHoriz,
                                Max((_v1 - _v0) * varPlotW, 0),
                                Max(_catBarFrac * varPlotW, 0)
                              ),
                              _bh: If(
                                varIsHoriz,
                                Max(_catBarFrac * varPlotH, 0),
                                Max((_v1 - _v0) * varPlotH, 0)
                              ),
                              _px: If(
                                varIsHoriz,
                                varPlotX + _pointFrac * varPlotW,
                                varPlotX + _catGroupMid * varPlotW
                              ),
                              _py: If(
                                varIsHoriz,
                                varPlotY + _catGroupMid * varPlotH,
                                varPlotY + varPlotH * (1 - _pointFrac)
                              )
                            },
                            {
                              SegKey:       _thisCat & "|" & _thisSeries,
                              Category:     _thisCat,
                              Series:       _thisSeries,
                              Value:        _thisVal,
                              DisplayValue: _thisVal,
                              CatIdx:       _catIdx,
                              SerIdx:       _serIdx,
                              Color:        _color,
                              DrawAs:       _drawAs,
                              MarkerType:   _marker,
                              BarX:         _bx,
                              BarY:         _by,
                              BarW:         _bw,
                              BarH:         _bh,
                              PointX:       _px,
                              PointY:       _py,
                              Tooltip:      _thisCat & " / " & _thisSeries & ": " & Text(_thisVal)
                            }
                          )
                        )
                      )
                    )
                  )
                )
              )
            )
          )
        );

        If(
          varSelectedKey <> "" And IsBlank(LookUp(varBarSegments, SegKey = varSelectedKey)),
          Set(varSelectedKey, "")
        );

        // ====

                  _sc:   If(_isMaj, "#CBD5E1", "#E5E7EB")
                },
                If(
                  varIsHoriz,
                  "<​line x1='" & Text(varPlotX + _frac * varPlotW, "[$-en-US]0.00") &
                  "' y1='" & Text(varPlotY, "[$-en-US]0.00") &
                  "' x2='" & Text(varPlotX + _frac * varPlotW, "[$-en-US]0.00") &
                  "' y2='" & Text(varPlotY + varPlotH, "[$-en-US]0.00") &
                  "' stroke='" & _sc & "' stroke-width='" & _sw & "'/>",
                  "<​line x1='" & Text(varPlotX, "[$-en-US]0.00") &
                  "' y1='" & Text(varPlotY + varPlotH * (1 - _frac), "[$-en-US]0.00") &
                  "' x2='" & Text(varPlotX + varPlotW, "[$-en-US]0.00") &
                  "' y2='" & Text(varPlotY + varPlotH * (1 - _frac), "[$-en-US]0.00") &
                  "' stroke='" & _sc & "' stroke-width='" & _sw & "'/>"
                )
              )
            )
          )
        );

        // 9b: Bar rectangles
        Set(
          varSvgBars,
          Concat(
            Sort(Filter(varBarSegments, DrawAs = "bar"), CatIdx * 1000 + SerIdx),
            "<​rect x='" & Text(BarX, "[$-en-US]0.00") &
            "' y='" & Text(BarY, "[$-en-US]0.00") &
            "' width='" & Text(Max(BarW, 0), "[$-en-US]0.00") &
            "' height='" & Text(Max(BarH, 0), "[$-en-US]0.00") &

        Set(
          varSvgLines,
          Concat(
            Filter(varSeriesList, SerDrawAs <> "bar"),
            "<polyline points='" &
            Concat(

        Set(
          varSvgMarkers,
          Concat(

                _mt: MarkerType

              If(
                _mt = "Circle",
                "<​circle cx='" & _mx & "' cy='" & _my & "' r='5' fill='white' stroke='" & _mc & "' stroke-width='2'/>",
                _mt = "FilledCircle",
                "<​circle cx='" & _mx & "' cy='" & _my & "' r='5' fill='" & _mc & "' stroke='white' stroke-width='1.5'/>",

                  _mx & "," & Text(PointY + 6, "[$-en-US]0.00") & " " &

                "' fill='" & _mc & "' stroke='white' stroke-width='1.5'/>",

                "' y='" & Text(_ty, "[$-en-US]0.00") &

            "' y1='" & Text(varPlotY, "[$-en-US]0.00") &

                _tickLen: If(_isMaj, 8, 4)

              If(
                varIsHoriz,
                "<​line x1='" & Text(varPlotX + _frac * varPlotW, "[$-en-US]0.00") &
                "' y1='" & Text(varPlotY + varPlotH, "[$-en-US]0.00") &

                "' stroke='#333' stroke-width='" & If(_isMaj, "1.5", "0.75") & "'/>" &
                If(
                  _isMaj,
                  "<​text x='" & Text(varPlotX + _frac * varPlotW, "[$-en-US]0.00") &
                  "' y='" & Text(varPlotY + varPlotH + 22, "[$-en-US]0.00") &

                  ""

                "<​line x1='" & Text(varPlotX - _tickLen, "[$-en-US]0.00") &
                "' y1='" & Text(varPlotY + varPlotH * (1 - _frac), "[$-en-US]0.00") &
                "' x2='" & Text(varPlotX, "[$-en-US]0.00") &
                "' y2='" & Text(varPlotY + varPlotH * (1 - _frac), "[$-en-US]0.00") &
                "' stroke='#333' stroke-width='" & If(_isMaj, "1.5", "0.75") & "'/>" &
                If(
                  _isMaj,
                  "<​text x='" & Text(varPlotX - 12, "[$-en-US]0.00") &
                  "' y='" & Text(varPlotY + varPlotH * (1 - _frac), "[$-en-US]0.00") &

                _slotStart: varPlotX + ((CatIdx - 1) / Max(varCatCount, 1)) * varPlotW,
                _estHalf:   Len(CatName) * 3.6 * Abs(Cos(varLabelAngle * Pi() / 180)) / 2
              },
              If(
                varIsHoriz,
                "<​line x1='" & Text(varPlotX - 6, "[$-en-US]0.00") &
                "' y1='" & Text(_tickYH, "[$-en-US]0.00") &
                "' x2='" & Text(varPlotX, "[$-en-US]0.00") &
                "' y2='" & Text(_tickYH, "[$-en-US]0.00") &
                "' stroke='#333' stroke-width='1'/>" &
                "<​text x='" & Text(varPlotX - 10, "[$-en-US]0.00") &
                "' y='" & Text(_tickYH, "[$-en-US]0.00") &
                "' text-anchor='end' dominant-baseline='middle' font-size='14' font-family='Segoe UI' fill='#333'>" &
                CatName &
                "<​/text>",
                // ---
                With(
                 {
                    _labelX:  _tickX,
                    _sinA:    Abs(Sin(varLabelAngle * Pi() / 180)),
                    _cosA:    Abs(Cos(varLabelAngle * Pi() / 180)),
                    _charH:   8
                  },
                  With(
                    {
                      _vertHalfExtent: (_estHalf * _sinA + _charH * _cosA) * 2
                    },
                    With(
                      {
                        _labelY: _tickY + 6 + _vertHalfExtent + 2
                      },
                      "<​line x1='" & Text(_tickX, "[$-en-US]0.00") &
                      "' y1='" & Text(_tickY, "[$-en-US]0.00") &
                      "' x2='" & Text(_tickX, "[$-en-US]0.00") &
                      "' y2='" & Text(_tickY + 6, "[$-en-US]0.00") &
                      "' stroke='#333' stroke-width='1'/>" &
                      "<text x='" & Text(_labelX, "[$-en-US]0.00") &
                      "' y='" & Text(_labelY, "[$-en-US]0.00") &
                      "' text-anchor='middle' font-size='14' font-family='Segoe UI' fill='#333'" &
                      If(
                        varLabelAngle <> 0,
                        " transform='rotate(" & Text(-varLabelAngle, "[$-en-US]0.00") & "," &

              )

        Set(
          varSvgAxisTitles,
          If(
            Coalesce(cmpBarChart.ValueAxisTitle, "") <> "",
            If(
              varIsHoriz,
              "<​text x='" & Text(varPlotX + varPlotW / 2, "[$-en-US]0.00") &

          If(
            Coalesce(cmpBarChart.CategoryAxisTitle, "") <> "",
            If(
              varIsHoriz,
              "<​text x='14' y='" & Text(varPlotY + varPlotH / 2, "[$-en-US]0.00") &
              "' text-anchor='middle' font-size='16' font-family='Segoe UI' font-weight='600' fill='#333'" &

        // 9i: Assemble full SVG
        Set(
          varSVG,
          If(
            Or(IsEmpty(varBarSegments), varCatCount = 0),

            varSvgBars &

            varSvgMarkers &
            varSvgValueLabels &
            varSvgValueAxis &
            varSvgCatAxis &
            varSvgAxisTitles &

        // SECTION 10: BUILD varOverlayRects
        // ====
        Set(varLineHitR, 18);

        ClearCollect(
          varOverlayRectsI,
          ForAll(
            Filter(varBarSegments, DrawAs = "bar"),
            With(
              {
                _px: (BarX - varPlotX) / varPlotW * varPlotPixW,

        =If(
